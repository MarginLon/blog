数字转中文

export function transformNumber(number) {
  const INDEX_MAP = ['零', '一'......]
  if(!number) return
  if(number === 10) return INDEX_MAP[number]
  return [...number.toString()].reduce((pre, cur) => pre + INDEX_MAP[cur], '')
}
